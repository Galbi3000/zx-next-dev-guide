    \begin{instruction}{JP nm} 
        \Symbol{\SymJP{nm}}
            \FlagsJPnn
            \OpCode{11}{000}{011}
            \Hex{C3}{3}
            \Cycles{3}{10}
            \Comment{
                \multirow{8}{*}{
                    \tt
                    \begin{tabular}{ll}
                        c & \OCT{c} \\
                        \hline
                        NZ & 000 \\
                        Z & 001 \\
                        NC & C10 \\
                        C & 011 \\
                        PO & 100 \\
                        PE & 101 \\
                        P & 110 \\
                        M & 111 \\
                    \end{tabular}
                }
            }
        \SkipToOpCode 
            \OpRange{m} 
            \Hex{..}{}
        \SkipToOpCode 
            \OpRange{n} 
            \Hex{..}{}
    \end{instruction}

\begin{notestable}
    \NoteItem{{\tt e} is a signed two-complement in the range -127, 129.}
    \NoteItem{{\tt e-2} in the opcode provides an effective number of {\tt PC+e} as {\tt PC} is incremented by two prior to the addition of {\tt e}.}
\end{notestable}
